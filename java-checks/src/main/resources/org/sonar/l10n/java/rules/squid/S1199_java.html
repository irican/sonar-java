嵌套代码块可用于创建新范围并限制其中定义的变量的可见性。在方法中使用此功能通常表示该方法具有太多功能，应该重构为较小的方法。

不合规的代码示例

public void evaluate(int operator) {
  switch (operator) {
    /* ... */
    case ADD: {                                // 不合规 - 嵌套代码块 '{' ... '}'
        int a = stack.pop();
        int b = stack.pop();
        int result = a + b;
        stack.push(result);
        break;
      }
    /* ... */
  }
}

合规的解决方法

public void evaluate(int operator) {
  switch (operator) {
    /* ... */
    case ADD:                                  // 合规
      evaluateAdd();
      break;
    /* ... */
  }
}

private void evaluateAdd() {
  int a = stack.pop();
  int b = stack.pop();
  int result = a + b;
  stack.push(result);
}
